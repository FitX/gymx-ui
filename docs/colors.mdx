import { Meta, ColorItem, ColorPalette } from '@storybook/blocks';
import { getCSSCustomProps } from './demo/utils/css-props';

export const accentColors = getCSSCustomProps('--gymx-color-accent');
export const grayColors = getCSSCustomProps('--gymx-color-black');
export const generateItems = (items = []) => items.map(([name, value], index) => <ColorItem key={index} title={name} subtitle="" colors={{ [name]: value}} />);

export const AccentColorItems = () => generateItems(accentColors);
export const GrayColorItems = () => generateItems(grayColors);
